    ![cadow (11).jpg](/images/cadow\(11\).jpg)

  <Step title="Select the Type of your offer. There are three types of offers:">
    - Club Offer
    - Bundle Offer
    - Flash Sale

    ![cadow (2).jpg](/images/cadow\(2\).jpg)
  </Step>
  <Step title="Select Bundle Offer from the drop-down, enter a title (e.g., *Fitness Bundle*), and optionally set an expiry date to limit the offer's availability.">
    ![cadow (10).jpg](/images/cadow\(10\).jpg)
  </Step>
  <Step title="Click on Add Product, choose the product for your offer, select the variant, set the price and quantity, then click Add Product.">
    ![cadow (4).jpg](/images/cadow\(4\).jpg)
  </Step>
  <Step title="Repeat this step until you create all the products you would like to have in your offer. Once finished click on Create offer.">
    ![cadow (1).jpg](/images/cadow\(1\).jpg)
  </Step>
  <Step title="A green notification will appear confirming your offer has successfully been created.">
    ![cadow (5).jpg](/images/cadow\(5\).jpg)
  </Step>
  <Step title="Click on Discounts in the menu to create a discount you would like to apply to your bundle offer. If you need more detailed instructions, please refer to the Creating a Discount section.">
    
  </Step>
  <Step title="Click on your offer and click on Promotions tab. Then click on Add Discount.">
    ![cadow (6).jpg](/images/cadow\(6\).jpg)
  </Step>
  <Step title="Chose the promotion you would like to apply to your bundle offer and click on Add.">
    ![cadow (7).jpg](/images/cadow\(7\).jpg)
  </Step>
  <Step title="Click on Add Selected.">
    ![cadow (8).jpg](/images/cadow\(8\).jpg)
  </Step>
