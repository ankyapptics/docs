---
title: "List orders"
description: "Get all orders for the authenticated account with filtering, sorting, and pagination"
api: "POST https://crm.apptics.ai/api/public/v1/orders/list"
authMethod: "bearer"
---

### Authorizations

<ParamField header="Authorization" type="string" required>
  Enter your API key as: `Bearer your-api-key`
</ParamField>

/

### Body

<ParamField body="search" type="string">
  Search term to filter orders
</ParamField>

<ParamField body="sortBy" type="object">
  Sorting configuration
</ParamField>

<ParamField body="pagination" type="object">
  <Expandable title="properties">
    <ParamField body="page" default="1" type="integer">
      The page number to retrieve
    </ParamField>

    <ParamField body="pageSize" default="20" type="integer">
      The number of items per page
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="filters" type="object">
  Filtering criteria to narrow down results
</ParamField>

### Response

<ResponseField name="items" type="Order object[]">
  <Expandable title="Order">
    <ResponseField name="id" type="string">
      The unique identifier for the order
    </ResponseField>

    <ResponseField name="status" type="string">
      Current status of the order (e.g., paid)
    </ResponseField>

    <ResponseField name="createdAt" type="string">
      Timestamp of order creation
    </ResponseField>

    <ResponseField name="customer" type="object">
      <Expandable title="Customer Details">
        <ResponseField name="email" type="string">
          Customer email address
        </ResponseField>

        <ResponseField name="name" type="string">
          Customer full name
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="total" type="integer">
      Total amount in cents
    </ResponseField>

    <ResponseField name="currency" type="string">
      Currency code (e.g., USD)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer">
  Total number of orders matching the criteria
</ResponseField>

<ResponseField name="pageCount" type="integer">
  Total number of available pages
</ResponseField>
