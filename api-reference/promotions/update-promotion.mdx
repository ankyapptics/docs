### Request Body

<ParamField body="id" type="string" required>
  The promotion ID to update.
</ParamField>

<ParamField body="data" type="object" required>
  Updated promotion data.
  <Expandable title="properties" defaultOpen={true}>
    <ParamField body="storeId" type="string" required>
      The store ID to update the promotion for.
    </ParamField>
    <ParamField body="name" type="string" required>
      Name of the promotion.
    </ParamField>
    <ParamField body="code" type="string">
      Coupon code for the promotion (`null` for automatic promotions).
    </ParamField>
    <ParamField body="automatic" type="boolean">
      Whether the promotion applies automatically without a code.
    </ParamField>
    <ParamField body="enabled" type="boolean">
      Whether the promotion is active.
    </ParamField>
    <ParamField body="usageLimit" type="number">
      Maximum number of times the promotion can be used (`null` for unlimited).
    </ParamField>
    <ParamField body="startDate" type="string">
      Start date of the promotion (ISO 8601 format).
    </ParamField>
    <ParamField body="endDate" type="string">
      End date of the promotion (ISO 8601 format).
    </ParamField>
    <ParamField body="ruleOperator" type="string">
      Logical operator applied between rules. Options: `AND`, `OR`.
    </ParamField>
    <ParamField body="combinesWithOrderLevelDiscounts" type="boolean">
      Whether this promotion can combine with order-level discounts.
    </ParamField>
    <ParamField body="combinesWithLineItemDiscounts" type="boolean">
      Whether this promotion can combine with line-item discounts.
    </ParamField>
    <ParamField body="combinesWithShippingDiscounts" type="boolean">
      Whether this promotion can combine with shipping discounts.
    </ParamField>
    <ParamField body="forceCombine" type="boolean">
      Force combination with other promotions regardless of settings.
    </ParamField>
    <ParamField body="rules" type="array">
      Conditions that must be met for the promotion to apply.
    </ParamField>
    <ParamField body="actions" type="array">
      Effects applied when the promotion conditions are met.
    </ParamField>
  </Expandable>
</ParamField>
