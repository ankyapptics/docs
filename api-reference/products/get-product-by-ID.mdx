> Create a new product with variants and prices for a specific store

### Body

<ParamField body="storeId" type="string" required>
  The unique identifier of the store.
</ParamField>

<ParamField body="name" type="string" required>
  The name of the product.
</ParamField>

<ParamField body="description" type="string">
  The description of the product.
</ParamField>

<ParamField body="unitLabel" type="string">
  Label for the unit (e.g., "bottle", "box").
</ParamField>

<ParamField body="accountingCode" type="string">
  Accounting code for the product.
</ParamField>

<ParamField body="active" type="boolean">
  Whether the product is active.
</ParamField>

<ParamField body="isShippable" type="boolean">
  Whether the product requires shipping.
</ParamField>

<ParamField body="isTaxable" type="boolean">
  Whether the product is taxable.
</ParamField>

<ParamField body="autoSync" type="boolean">
  Whether to automatically sync the product.
</ParamField>

<ParamField body="taxCategory" type="string">
  The tax category for the product.
</ParamField>

<ParamField body="variants" type="array" required>
  List of product variants.
  <Expandable title="items">
    <ParamField body="name" type="string" required>
      Variant name.
    </ParamField>
    <ParamField body="description" type="string">
      Variant description.
    </ParamField>
    <ParamField body="sku" type="string">
      Stock Keeping Unit.
    </ParamField>
    <ParamField body="imageUrl" type="string">
      URL of the variant image.
    </ParamField>
    <ParamField body="grams" type="number">
      Weight in grams.
    </ParamField>
    <ParamField body="active" type="boolean">
      Whether the variant is active.
    </ParamField>
    <ParamField body="default" type="boolean">
      Whether this is the default variant.
    </ParamField>
    <ParamField body="prices" type="array" required>
      Pricing configurations for this variant.
      <Expandable title="items">
        <ParamField body="currencyOptions" type="object" required>
          Map of currency codes to price details (amount).
        </ParamField>
        <ParamField body="recurring" type="boolean" required>
          Is this a recurring price?
        </ParamField>
        <ParamField body="billingTiming" type="string" required>
          Billing timing (e.g., "usage", "in_advance").
        </ParamField>
        <ParamField body="interval" type="string" required>
          Billing interval (e.g., "month").
        </ParamField>
        <ParamField body="intervalCount" type="integer" required>
          Count of intervals.
        </ParamField>
        <ParamField body="default" type="boolean">
          Is this the default price?
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>
